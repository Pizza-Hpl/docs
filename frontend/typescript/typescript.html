# 让vscode自动编译tsc文件

  1. 设置vscode自动编译

    1. 运行tsc --init , 创建tsconfig.json文件

      输入命令:tsc --init
会生成一个tsconfig.json文件

    2. 修改tsconfig.json文件,设置js文件夹: "outDir":"./js/"

      在tsconfig.json中找到outDir,取消注释,修改要编译文件夹路径,如: "outDir": "./js/",

    3. 设置vscode监视任务

      vscode顶部选择终端---运行任务---选中监视tsconfig.json---后面项目会生成一个js文件夹,编译的文件都在里面

# tuple元祖

概念:就是一个规定了元素数量和每个元素类型的"数组"而每个元素的类型,可以不相同

特点:

  1. 声明时要指定元素个数
  2. 声明时,要为每个元素规定类型

语法

let 元祖名:[类型1,类型2,类型3] = [值1,值2,值3]
let tup:[string.number,boolean] = ['张三', 24, true]

注意:值得类型一定要跟声明的类型一致,不然报错

代码

let tup:[string,number,boolean] = ['张三',15,true]

tup = ['李四',18,true] //再次赋值也要跟声明的类型一致

// 访问 元祖中的元素 和长度
console.log(tup[1])//下标
console.log(tup.length)

# enum枚举

声明语法

enum 枚举名{
枚举项1 = 枚举值1,
枚举项1 = 枚举值1
...     ....
}

如:enum Gun {
M416 = 1,
AK47 = 2,
Goza = 3
}

枚举项一般用英文和数字 , 而 枚举值用整型数字

使用默认枚举值

enum 枚举名{
枚举项1 ,
枚举项1 
...     ....
}

如:enum Gun {
M416,  //-> 0
AK47 , //-> 1
Goza   //-> 2
}

枚举值将自动生成从 0 开始的数值

enum Gender{
    boy =1,
    girl =2,
    unknown =3
}

let usrSex:Gender = Gender.boy; //声明变量等于Gender.boy
if (usrSex == Gender.boy) {  //如果 usrSex 等于 Gender.boy  1
    console.log("你是男的");//就打印男的
    
} else {
    console.log("你是女的或不知性别"); //不等于1,就打印女或其他  2 or 3
    
}

打印结果: 你是男的

代码编译

// 枚举 --- 判断性别
var Gender;
(function (Gender) {
    Gender[Gender["boy"] = 1] = "boy";
    Gender[Gender["girl"] = 2] = "girl";
    Gender[Gender["unknown"] = 3] = "unknown";
})(Gender || (Gender = {}));
var usrSex = Gender.boy; //声明变量等于Gender.boy
if (usrSex == Gender.boy) { //如果 usrSex 等于 Gender.boy  1
    console.log("你是男的"); //就打印男的
}
else {
    console.log("你是女的或不知性别"); //不等于1,就打印女或其他  2 or 3
}

# never不存在的值得类型

**概念:**never 代表不存在的值得类型,常用作为抛出异常或者无线循环的函数返回类型

语法

function test():naver{

	}

# 函数的返回值和参数

函数 返回值类型

function 函数名():返回值类型{ //如果函数没有返回值,则定义为void
	
}

let 变量名:变量类型 = 函数名();

代码

函数 形参类型

function 函数名(形参1:类型 , 形参2:类型):返回值类型{

}
let 变量名:变量类型 = 函数名(实参1,实参2)


特点:
实参 和 形参 的类型要一致
实参 和 形参 的数量要一致

代码

    console.log(citName);    
}

Cit('G镇')//

// 注意: ts中 实参和形参的数量必须一致

# 函数的可选参数

函数 可选参数

function 函数名(形参 ? : 类型):返回值类型{

}

注:形参 ? 指的是可选参数的实参可传,也可不传

调用:
可以不传递实参   函数名();
可以传递实参     函数名(实参值);

# 函数的默认值

函数的默认值

function 函数名(形参1:类型 = 默认值1, 形参2:类型 = 默认值2):返回值类型{
    
}

注:带默认值的参数 本身也是可选参数

调用

不传递实参	函数名();	--->编译后		函数名(默认值1,默认值2);

传递1个实参	函数名(实参1);	--->编译后		函数名(实参1,默认值2);

传递2个实参	函数名(实参1,实参2);	--->编译后		函数名(实参1,实参2);

只传第2个实参	函数名(undefined,实参2);	--->编译后		函数名(默认值1,实参2);

# 函数的剩余参数

问题

function add(a : number , b:number):void{
    console.log(a+b);
}

注:如果add()里面还有更多不确定的参数,c , d ... 就要用到剩余参数了

剩余参数

function add(形参1:类型 , 形参2:类型, ...形参3:类型[]):void{
    console.log(a+b)
}

语法: ...形参:类型[]
注意:剩余函数一定要放在函数的最后面

//需求:把剩余参数求和

function add(a:number, b:number,...cdeNum:number[]) {
    //创建一个 求和变量 保存求和结果, 将 a b 的值求和后 存入
    let addNum:number = a+b;
    //使用 for of 语法遍历剩余参数 数组中 每个元素,并
    for (let ele of cdeNum) {
        addNum += ele
    }
    //打印结果
    console.log('打印结果:' + addNum);
    
}

add(1,2)//a和b
add(1,2,3,4,4,5,6,7,8)//1和2是a,b,其它参数都是cdeNum

// 打印结果
// 打印结果:3
// 打印结果:40

编译成javascript中的代码

//需求:把剩余参数求和
function add(a, b) {
    var cdeNum = [];
    for (var _i = 2; _i < arguments.length; _i++) {
        cdeNum[_i - 2] = arguments[_i];
    }
    //创建一个 求和变量 保存求和结果, 将 a b 的值求和后 存入
    var addNum = a + b;
    //使用 for of 语法遍历剩余参数 数组中 每个元素,并累加到求和变量中
    for (var _a = 0, cdeNum_1 = cdeNum; _a < cdeNum_1.length; _a++) {
        var ele = cdeNum_1[_a];
        addNum += ele;
    }
    //打印结果
    console.log('打印结果:' + addNum);
}
add(1, 2); //a和b
add(1, 2, 3, 4, 4, 5, 6, 7, 8); //1和2是a,b,其它参数都是cdeNum
// 打印结果
// 打印结果:3
// 打印结果:40

特点:

  1. 剩余参数 只能 定义有一个
  2. 剩余参数 只能 定义为数组
  3. 剩余参数 只能 定义在 形参列表最后

调用:

  1. 传递2个实参 函数名(1,2)
  2. 传递3个实参 函数名(1,2,4)
  3. 传递2+个实参 函数名(1,2,3,4,5,6,....)

# localStroage操作

  • localStorage 用于在在浏览器端持久化保存键值对数据

  • localStorage 特点

    • 大小限制:5M(chrome浏览器)更大数据可以使用浏览器本地数据库(indexDB或webSpl)
    • 受同源访问限制,不允许跨域访问
    • 在浏览器隐私模式下无法使用
    • 因为在本地保存,不会发送数据,网络爬虫无法抓取
    • 只能存放字符串

  • localStorage 基本语法

    方法名 作用
    localStorage.setItem('key','value') 存放 键值对 数据
    localStorage.getltem('key') 获取 根据key 查询 value值,没有则返回 null
    localStorage.removeltem('key') 删除 根据key删除 对应 键值对
    localStorage.clear() 清空所有键值对 数据

  • localStorage 读写 对象

    • 保存

      //1,先将对象转成JSON字符串,然后再保存
let strJson:string = JSON.stringify(对象);

//2,保存json字符串到本地
localStorage.setltem('key',strJson)

    • 读取

      //1,取出json字符串
let strJson:string | null = localStorage.getltem('key')

//2,将json字符串转成对象
let obj = JSON.parse(strJson as string)

//0,获取 文本框对象 any不确定查询的类型是什么类型所以就用any
let txtkeyObj: any = document.querySelector('#txtkey');
let txtValueObj: any = document.querySelector('#txtvalue');
let btnAdd: any = document.querySelector('#btnAdd'); // 1,保存数据
let btnGet: any = document.querySelector('#btnGet');// 2,取数据
let btnRemover: any = document.querySelector('#btnRemover');// 3,删除数据
let btnClear: any = document.querySelector('#btnClear');// 4,清空数据

let btnAddObj: any = document.querySelector('#btnAddObj');//5,保存对象
let btnGetObj: any = document.querySelector('#btnGetObj');//6,取对象

// 保存数据
btnAdd.onclick = function(){
    // 获取文本框内容
    let Addkey = txtkeyObj.value; 
    let Addvalue = txtValueObj.value;

    // 将两个字符串数据 保存 到 本地硬盘中
    localStorage.setItem(Addkey,Addvalue)
    alert('保存成功')
}

// 2,取数据
btnGet.onclick = function(){
    let Addkey = txtkeyObj.value;
    // 从本地数据中 根据 key 取出 value值
    let Addvalue = localStorage.getItem(Addkey);
    alert('取出的数据是'+ Addvalue);
    
}

// 3,删除数据
btnRemover.onclick = function(){
    let Addkey = txtkeyObj.value;
    // 从本地数据中 根据 key 删除 value值
    localStorage.removeItem(Addkey)
    alert('你删除的是'+ Addkey)
}

// 4,清空数据
btnClear.onclick = function(){
    localStorage.clear()
}


//5,保存对象
btnAddObj.onclick = function(){
    let dog:object = {
        name:'哈士奇',
        age:2,
        gender:true
    };

    // 先将对象 转成JSON字符串,然后再保存
    let strJson:string = JSON.stringify(dog);
    //保存 json字符串 到本地
    localStorage.setItem('myDog',strJson);
    alert('保存成功')
}

// 6,取对象
btnGetObj.onclick = function(){
    // 1,取出json字符串
    let strJson:string | null = localStorage.getItem('myDog');
    // 2,将json字符串 转成 对象
    let obj = JSON.parse(strJson as string);
    console.log(obj)
}
